<script>
    // Funció per calcular els costos automàticament
    function calcular() {
        let totalGeneral = 0;
        const files = document.querySelectorAll("#taulaEscandall tbody tr");

        files.forEach(fila => {
            const qty = parseFloat(fila.querySelector(".qty").value) || 0;
            const preu = parseFloat(fila.querySelector(".preu").value) || 0;
            
            // Suposem que el preu és per kg/l (dividim qty per 1000)
            const cost = (qty / 1000) * preu;
            
            fila.querySelector(".cost-fila").innerText = cost.toFixed(2) + " €";
            totalGeneral += cost;
        });

        document.getElementById("totalFinal").innerText = totalGeneral.toFixed(2);
    }

    // Funció per afegir noves files sense trencar el codi
    function afegirFila() {
        const tbody = document.querySelector("#taulaEscandall tbody");
        const novaFila = document.createElement("tr");
        novaFila.innerHTML = `
            <td><input type="text" placeholder="Ingredient" class="nom"></td>
            <td><input type="number" value="0" class="qty" oninput="calcular()"></td>
            <td><input type="number" value="0" class="preu" oninput="calcular()"></td>
            <td class="cost-fila">0.00 €</td>
        `;
        tbody.appendChild(novaFila);
    }

    // Funció per generar PDF (Corregida per a jsPDF modern)
    async function generarPDF() {
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF();
        
        doc.text("Resum de l'Escandall", 10, 10);
        let y = 20;
        
        const files = document.querySelectorAll("#taulaEscandall tbody tr");
        files.forEach(fila => {
            const nom = fila.querySelector(".nom").value || "Sense nom";
            const cost = fila.querySelector(".cost-fila").innerText;
            doc.text(`${nom}: ${cost}`, 10, y);
            y += 10;
        });

        doc.text(`TOTAL: ${document.getElementById("totalFinal").innerText} €`, 10, y + 10);
        doc.save("escandall.pdf");
    }
</script>
